function searchPlayer(playerName) { Papa.parse('all_players.csv', { download: true, header: true, complete: function(allPlayersData) { const player = allPlayersData.data.find(p => p['Player Name'].toLowerCase() === playerName.toLowerCase()); if (player) { $('#years-active').text(player.Years); let years = player.Years.match(/\d{4}/g); let startYear = parseInt(years[0]); let endYear = parseInt(years[1]); // New function to parse baseball data and display it function parseAndDisplayData(file) { Papa.parse(file, { download: true, header: true, complete: function(baseballData) { $("#yearly-ranks").empty(); for(let year = startYear; year <= endYear; year++) { const playersOfYear = baseballData.data.filter(p => p.Year == year); playersOfYear.sort((a, b) => parseFloat(b.WAR) - parseFloat(a.WAR)); const top30Players = playersOfYear.slice(0, 30).map(p => p['Player Name'].split(' ').slice(-1).join(' ')); let columnHtml = `
${year}
`; for(const player of top30Players) { columnHtml += `${player}
`; } columnHtml += '
'; $("#yearly-ranks").append(columnHtml); } } }); } // Check in both files Papa.parse('baseball_pitchers_1900_2024.csv', { download: true, header: true, complete: function(pitcherData) { const pitcher = pitcherData.data.find(p => p['Player Name'].toLowerCase() === playerName.toLowerCase()); if (pitcher) { parseAndDisplayData('baseball_pitchers_1900_2024.csv'); } else { Papa.parse('baseball_hitters_1900_2024.csv', { download: true, header: true, complete: function(hitterData) { const hitter = hitterData.data.find(p => p['Player Name'].toLowerCase() === playerName.toLowerCase()); if (hitter) { parseAndDisplayData('baseball_hitters_1900_2024.csv'); } } }); } } }); } } }); }
